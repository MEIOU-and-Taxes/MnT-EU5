	map_color = {
		if = {
			limit = {
				has_owner = yes
				has_variable = mnt_location_pop_wealth_total
			}
			if = {
				limit = { var:mnt_location_pop_wealth_total < 0 }
				lerp = {
					min_color = define:NMapColors|MAP_COLOR_LOW
					max_color = define:NMapColors|MAP_COLOR_MIN
					factor = {
						value = var:mnt_location_pop_wealth_total
						divide = global_var:mnt_location_pop_wealth_total_max
						multiply = mnt_location_pop_wealth_threshold_low_inverse
						multiply = -1
					}
				}
			}
			else_if = {
				limit = { var:mnt_location_pop_wealth_total > mnt_location_pop_wealth_high }
				lerp = {
					min_color = define:NMapColors|MAP_COLOR_HIGH
					max_color = define:NMapColors|MAP_COLOR_TOP
					factor = {
						value = var:mnt_location_pop_wealth_total
						divide = global_var:mnt_location_pop_wealth_total_max
						subtract = mnt_location_pop_wealth_threshold_high
						multiply = mnt_location_pop_wealth_multiplier_high
					}
				}
			}

					max_color = define:NMapColors|MAP_COLOR_HIGH

						value = var:mnt_location_pop_wealth_total
