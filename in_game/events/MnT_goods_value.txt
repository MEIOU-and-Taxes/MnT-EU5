namespace = mnt_goods_value

# Monthly recalc of total goods value (urban + rural) for each country
mnt_goods_value.1 = {
	type = country_event
	title = mnt_goods_value.1.title
	hidden = yes
	trigger = {
		is_ai = no
		OR = {
			NOT = { has_variable = mnt_goods_value_update_freq }
			var:mnt_goods_value_update_freq = 0
		}
	}

	immediate = {
		if = {
			limit = { NOT = { has_variable = mnt_goods_value_update_freq } }
			set_variable = { name = mnt_goods_value_update_freq value = 0 }
		}
		calc_country_goods_value_produced = yes
	}
}

# Yearly recalc of total goods value (urban + rural) for each country
mnt_goods_value.2 = {
	type = country_event
	title = mnt_goods_value.1.title
	hidden = yes
	trigger = {
		is_ai = no
		has_variable = mnt_goods_value_update_freq
		var:mnt_goods_value_update_freq = 1
	}

	immediate = {
		calc_country_goods_value_produced = yes
	}
}
