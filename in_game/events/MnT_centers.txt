namespace = mnt_centers

# Yearly recalculation of centers (trade / production / culture / education)
mnt_centers.1 = {
	type = country_event
	title = hidden_event.t
	hidden = yes

		# Clear previous center modifiers
		every_location_in_the_world = {
			limit = {
				# By checking the conditions here, later we can simply check for the existence of the variable
				is_land = yes
				has_owner = yes
			}
			remove_location_modifier = mnt_center_trade_local
			remove_location_modifier = mnt_center_trade_regional
			remove_location_modifier = mnt_center_trade_continental
			remove_location_modifier = mnt_center_trade_world

			remove_location_modifier = mnt_center_production_continental

			remove_location_modifier = mnt_center_culture_continental

			remove_location_modifier = mnt_center_education_continental
			remove_location_modifier = mnt_center_education_world

			# Production heuristic: urban manufactured goods output per head
			save_scope_as = target_location
			calc_urban_goods_value_produced_location = yes

			# Cache scores to avoid recalculating during location iteration
			calc_center_trade_score_location = yes
			calc_center_production_score_location = yes
			calc_center_culture_score_location = yes
			calc_center_education_score_location = yes
		}

		# Trade Centers
		remove_global_variable = world_center
		every_continent = {
			remove_global_variable = continent_center

			every_sub_continent_in_continent = {
				every_region_in_continent = {
					remove_global_variable = region_center

					every_area_in_region = {
						ordered_location_in_area = {
							limit = {
								has_variable = mnt_center_trade_score_location
								this.var:mnt_center_trade_score_location >= center_trade_local_threshold
							}
							order_by = { value = this.var:mnt_center_trade_score_location }
							max = 1
							check_range_bounds = no
							add_location_modifier = { modifier = mnt_center_trade_local days = -1 }
							if = {
								limit = {
									this.var:mnt_center_trade_score_location > center_trade_regional_threshold
									OR = {
										NOT = { has_global_variable = region_center }
										this.var:mnt_center_trade_score_location > global_var:region_center.var:mnt_center_trade_score_location
									}
								}
								set_global_variable = {
									name = region_center
									value = this
								}
							}
						}
					}
					global_var:region_center ?= {
						remove_location_modifier = mnt_center_trade_local
						add_location_modifier = { modifier = mnt_center_trade_regional days = -1 }

						if = {
							limit = {
								this.var:mnt_center_trade_score_location >= center_trade_continental_threshold
								OR = {
									NOT = { has_global_variable = continent_center }
									this.var:mnt_center_trade_score_location > global_var:continent_center.var:mnt_center_trade_score_location
								}
							}
							set_global_variable = {
								name = continent_center
								value = this
							}
						}
					}
				}
			}
			
			global_var:continent_center ?= {
				remove_location_modifier = mnt_center_trade_regional
				add_location_modifier = { modifier = mnt_center_trade_continental days = -1 }

				if = {
					limit = {
						# World centers (late game, very scarce)
						current_age_or_later = { age = age_5_absolutism }
						this.var:mnt_center_trade_score_location >= center_trade_world_threshold
						OR = {
							NOT = { has_global_variable = world_center }
							this.var:mnt_center_trade_score_location > global_var:world_center.var:mnt_center_trade_score_location
						}
					}
					set_global_variable = {
						name = world_center
						value = this
					}
				}
			}
		}
		global_var:world_center ?= {
			remove_location_modifier = mnt_center_trade_continental
			add_location_modifier = { modifier = mnt_center_trade_world }
		}

							set_global_variable = {
								name = continent_center

							}

					}

					}

						name = world_center
						value = this
					}
