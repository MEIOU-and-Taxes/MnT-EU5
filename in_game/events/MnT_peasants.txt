namespace = peasant_villages

#This event is designed to look for markets that need more food and try to ramp up peasant food production. It's not perfect, since wars can spike demand badly.
peasant_villages.1 = {
	type = country_event
	title = peasant_villages.1.t
	desc = peasant_villages.1.d
	
	trigger = {
		country_exists_and_can_own_locations = yes
		is_ai = yes
		any_market_present_in_country = {
			market_monthly_food_balance < 0
			market_food_percentage < 60 #a check to make sure this isn't just a monthly weird war spike.
		}
	}
	
	hidden = yes
	
	immediate = {
		if = {
			limit = { monthly_income_total > 15 }
			every_market_present_in_country = {
				limit  = { market_monthly_food_balance < 0 }
				random_location_in_market = {
					limit  = { 
						owner = root
						can_build_building = building_type:farming_village
						location_unemployed_population_for_building_type = {
							building_type = building_type:farming_village
							value >= 5000
						}
						
					}
					construct_building = {
						building_type = building_type:farming_village
					}
				}
				if = {
					limit  = { market_monthly_food_balance < -50 }
					random_location_in_market = {
						limit  = { 
							owner = root
							can_build_building = building_type:farming_village
							location_unemployed_population_for_building_type = {
								building_type = building_type:farming_village
								value >= 4000 #it's ok if this partially employs, better than nothing
							}
							
						}
						construct_building = {
							building_type = building_type:farming_village
						}
					}
				}
			}
		}
		else = {
			every_market_present_in_country = {
				limit  = { market_monthly_food_balance < 0 }
				random_location_in_market = {
					limit  = { 
						owner = root
						can_build_building = building_type:farming_village
						location_unemployed_population_for_building_type = {
							building_type = building_type:farming_village
							value >= 4000 #it's ok if this partially employs, better than nothing
						}
						construct_building = {
							building_type = building_type:farming_village
							cost_multiplier = 0
						}
					}
				}
			}
		}
	}
}

#This event looks for markets with large amounts of food and frees up peasants in a random location without a large peasant surplus. This leaves the peasants free to migrate, be employed in other places, etc.
peasant_villages.2 = {
	type = country_event
	title = peasant_villages.2.t
	desc = peasant_villages.2.d
	
	trigger = {
		country_exists_and_can_own_locations = yes
		is_ai = yes
		any_market_present_in_country = {
			market_monthly_food_balance > 600 #much safer check, since there are never positive spikes of food, only negative.
		}
	}
	
	hidden = yes
	
	immediate = {
		every_market_present_in_country = {
			limit  = { market_monthly_food_balance > 600 }
			random_location_in_market = {
				limit = {
					owner = root
					has_building_with_at_least_one_level = farming_village
					location_unemployed_population_for_building_type = {
						building_type = building_type:farming_village
						value < 3000
					}
				}
				destroy_building = "building(building_type:farming_village|owner)"
			}
			random_location_in_market = {
				limit = {
					owner = root
					has_building_with_at_least_one_level = farming_village
					location_unemployed_population_for_building_type = {
						building_type = building_type:farming_village
						value < 3000
					}
				}
				destroy_building = "building(building_type:farming_village|owner)"
			}
		}	
	}
}
