namespace = mnt_location_wealth

# Hidden event to update location wealth pools each month.
mnt_location_wealth.1 = {
	type = country_event
	title = mnt_location_wealth.1.title
	hidden = yes

	immediate = {
		# Update Wealth Pools of the Estates in the Country
		mnt_update_estate_wealth_pools = yes

		# Sum Total Country Estate Local Power for Distribution
		mnt_country_pop_income_proportion_sum = yes

		every_owned_location = {
			# Initialize Location Pop Wealth Variables if not present
			mnt_init_location_pop_wealth_variables = yes

			# Update Location Wealth Pools based on Estate Power and Pop Types
			mnt_update_location_wealth_pools = yes
		}
	}
}
