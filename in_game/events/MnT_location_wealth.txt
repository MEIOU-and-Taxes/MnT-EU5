namespace = mnt_location_wealth

# Hidden event to update location wealth pools each month.
mnt_location_wealth.1 = {
	type = country_event
	title = mnt_location_wealth.1.title
	hidden = yes

	trigger = { country_exists_and_can_own_locations = yes }

	immediate = {
		update_location_wealth = yes
	}
}

# Hidden event to update location wealth pools at game start
mnt_location_wealth.2 = {
	type = country_event
	title = mnt_location_wealth.2.title
	hidden = yes

	immediate = {
		every_country = {
			limit = { country_exists_and_can_own_locations = yes }
			
			# Initialize global variables, such as maximum location wealth
			mnt_init_global_vars = yes
			update_location_wealth = yes
		}
	}
}
