namespace = LOGGER_CHARTS

# Enable logging yearly non-country information
LOGGER_CHARTS.01 = {
	type = country_event
	title = LOGGER_CHARTS.01.title
	hidden = yes
	orphan = yes
	immediate = {
		set_global_variable = is_logging_yearly_non_country_information
		error_log = "Enabled: Logging yearly non-country information"
	}
}

# Continue logging
LOGGER_CHARTS.02 = {
    type = country_event
	title = LOGGER_CHARTS.01.title
    hidden = yes
    immediate = {
        run_logging = yes
    }
}

# Disable logging yearly non-country information
LOGGER_CHARTS.04 = {
	type = country_event
	title = LOGGER_CHARTS.01.title
	hidden = yes
	orphan = yes
	immediate = {
		remove_global_variable = is_logging_yearly_non_country_information
		error_log = "Disabled: Logging yearly non-country information"
	}
}

# Enable logging yearly country information
LOGGER_CHARTS.05 = {
	type = country_event
	title = LOGGER_CHARTS.01.title
	hidden = yes
	orphan = yes
	immediate = {
		set_global_variable = is_logging_yearly_country_information
		error_log = "Enabled: Logging yearly country information"
	}
}

# Disable logging yearly country information
LOGGER_CHARTS.06 = {
	type = country_event
	title = LOGGER_CHARTS.01.title
	hidden = yes
	orphan = yes
	immediate = {
		remove_global_variable = is_logging_yearly_country_information
		error_log = "Disabled: Logging yearly country information"
	}
}
