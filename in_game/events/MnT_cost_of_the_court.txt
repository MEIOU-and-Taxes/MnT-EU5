namespace = cost_of_the_court

# Recalculate demands from cost of the court
cost_of_the_court.1 = {
	type = country_event
	title = hidden_event.t
	hidden = yes
	immediate = {
		capital.market.location = {
			# Initialize the variable if not exists
			if = {
				limit = {
					not = { has_variable = total_cost_of_the_court }
				}
				set_variable = {
					name = total_cost_of_the_court
					value = 0
				}
			}
			change_variable = {
				name = total_cost_of_the_court
				add = {
					value = "root.court_maintenance"
                    multiply = 0.1      #Mepper: Old value broke the economy, demand for 10% of slider
					divide = {
						value = 1
						add = "root.modifier:court_spending_efficiency"
						min = 0.01		#DBZ prevention
					}
					multiply = "root.country_economical_base"
				}
			}
		}
	}
}

# Aggregate all CotC demands into temporary market demands
cost_of_the_court.2 = {
	type = country_event
	title = hidden_event.t
	hidden = yes
	trigger = {
		tag = FRA
	}
	immediate = {
		every_market_in_world = {
			# Calculate current price corresponding to 1 unit of the demand basket
			set_local_variable = {
				name = demand_basket_price
				value = {
					value = 0
					add = {
						# 1 furniture
						value = 1
						multiply = "market_price(goods:furniture)"
					}
					add = {
						# 0.25 jewelry
						value = 0.25
						multiply = "market_price(goods:jewelry)"
					}
					add = {
						# 0.5 glass
						value = 0.5
						multiply = "market_price(goods:glass)"
					}
					add = {
						# 1 fine_cloth
						value = 1
						multiply = "market_price(goods:fine_cloth)"
					}
                    add = {
                        # 1 cloth
                        value = 1
                        multiply = "market_price(goods:cloth)"
                    }
                    add = {
                        # 1 leather
                        value = 1
                        multiply = "market_price(goods:leather)"
                    }
                    add = {
                        # 1 weaponry
                        value = 1
                        multiply = "market_price(goods:weaponry)"
                    }
                    add = {
                        # 0.5 books
                        value = 0.5
                        multiply = "market_price(goods:books)"
                    }
                    add = {
                        # 1 paper
                        value = 1
                        multiply = "market_price(goods:paper)"
                    }
				}
			}

			if = {
				# In the first month, there are ~10 markets that do not have this variable stored. Removing the condition leads to errors
				limit = { location = { has_variable = total_cost_of_the_court } }

				# Switching between two temporary demands
				# If we use only one temporary demand, the UI will not be updated properly
				if = {
					limit = {
						has_temporary_demand = demand:cost_of_the_court_demand1
					}
					remove_temporary_demand = demand:cost_of_the_court_demand1
					add_temporary_demand = {
						type = demand:cost_of_the_court_demand2
						months = -1
						scale = {
							value = "location.var:total_cost_of_the_court"
							divide = "local_var:demand_basket_price"
						}
					}
				}
                else = {
					remove_temporary_demand = demand:cost_of_the_court_demand2
					add_temporary_demand = {
						type = demand:cost_of_the_court_demand1
						months = -1
						scale = {
							value = "location.var:total_cost_of_the_court"
							divide = "local_var:demand_basket_price"
						}
					}
				}

				# Clear total CotC value from the last month
				location = {
					set_variable = {
						name = total_cost_of_the_court
						value = 0
					}
				}
			}
		}

	}
}
