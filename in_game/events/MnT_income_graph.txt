namespace = mnt_income_graph

mnt_income_graph.1 = {
	type = country_event
	title = hidden_event.t
	hidden = yes
	trigger = {
		is_ai = no
		OR = {
			NOT = { has_variable = mnt_income_graph_update_freq }
			var:mnt_income_graph_update_freq = 0
		}
	}

	immediate = {
		if = {
			limit = { NOT = { has_variable = mnt_income_graph_update_freq } }
			set_variable = { name = mnt_income_graph_update_freq value = 0 }
		}
		mnt_update_income_graph_history = yes
	}
}

mnt_income_graph.2 = {
	type = country_event
	title = hidden_event.t
	hidden = yes
	trigger = {
		is_ai = no
		has_variable = mnt_income_graph_update_freq
		var:mnt_income_graph_update_freq = 1
	}

	immediate = {
		mnt_update_income_graph_history = yes
	}
}
