namespace = mnt_food

mnt_food.1 = {
    type = country_event
    title = hidden_event.t
    hidden = yes

    immediate = {
		# This logic is to remove "added pops" that come by way of buildings introduced in setup. Buildings added in the setup create new pops rather than employing existing ones; this removes the equivalent count of peasants that were added
		every_owned_location = {
			save_temporary_scope_as = location_scope
			set_variable = {
				name = non_peasant_pops
				value = {
					add = num_pop_type:laborers
					add = num_pop_type:burghers
					add = num_pop_type:soldiers
					add = num_pop_type:clergy
					add = num_pop_type:nobles
				}
			}
			# Only bother with this if there were non-peasant pops to begin with
			if = {
				limit = {
					var:non_peasant_pops > 0
				}
				ordered_pop = {
					limit = {
						pop_type = pop_type:peasants
					}
					order_by = pop_size
					max = 1
					check_range_bounds = no
					save_temporary_scope_as = largest_peasant_pop
				}					
				set_variable = {
					name = largest_peasant_pop_size
					value = scope:largest_peasant_pop.pop_size
				}
				if = {
					limit = {
						var:largest_peasant_pop_size < var:non_peasant_pops
					}
					scope:largest_peasant_pop = {
						add_pop_size = {
							value = pop_size
							multiply = -1
						}
					}
					change_variable = {
						name = non_peasant_pops
						subtract = var:largest_peasant_pop_size
					}
					# Since the first pop group was too large, we need to iterate a second time
					ordered_pop = {
						limit = {
							pop_type = pop_type:peasants
						}
						order_by = pop_size
						max = 1
						check_range_bounds = no
						if = {
							limit = {
								pop_size < scope:location_scope.var:non_peasant_pops
							}
							add_pop_size = {
								value = pop_size
								multiply = -1
							}
						}
						else = {
							add_pop_size = {
								value = scope:location_scope.var:non_peasant_pops
								multiply = -1
							}
						}
					}
				}
				else = {
					scope:largest_peasant_pop = {
						add_pop_size = {
							value = scope:location_scope.var:non_peasant_pops
							multiply = -1
						}
					}
				}
			}
			# Cleanup
			remove_variable = largest_peasant_pop_size
			remove_variable = non_peasant_pops
		}
		every_owned_location = {
			limit = { 
				can_build_building = building_type:farming_village
				num_pop_type:peasants > 6
			}
			change_building_level_in_location = {
				building = building_type:farming_village
				value = {
					add = num_pop_type:peasants
					divide = 12 #one per 12000 peasants
					round = yes
				}
			}
		}    
    }
}

mnt_food.2 = {
	type = country_event
	title = hidden_event.t

	trigger = {
		country_exists_and_can_own_locations = yes
		is_ai = yes
		any_market_present_in_country = {
			market_monthly_food_balance < 0
			market_food_percentage < 80 #a check to make sure this isn't just a monthly weird war spike.
		}
	}
	
	hidden = yes
	
	immediate = {
		if = {
			limit = { monthly_income_total > 15 }
			every_market_present_in_country = {
				limit  = { 
					market_monthly_food_balance < 0
					market_food_percentage < 80
				}
				random_location_in_market = {
					limit  = { 
						owner ?= root
						can_build_building = building_type:farming_village
						location_unemployed_population_for_building_type = {
							building_type = building_type:farming_village
							value >= 4000 #it's ok if this partially employs, better than nothing
						}
						
					}
					construct_building = {
						building_type = building_type:farming_village
					}
				}
				if = {
					limit  = { market_monthly_food_balance < -50 }
					random_location_in_market = {
						limit  = { 
							owner ?= root
							can_build_building = building_type:farming_village
							location_unemployed_population_for_building_type = {
								building_type = building_type:farming_village
								value >= 3500 #it's ok if this partially employs, better than nothing
							}
							
						}
						construct_building = {
							building_type = building_type:farming_village
						}
					}
				}
			}
		}
		else = {
			every_market_present_in_country = {
				limit  = { market_monthly_food_balance < 0 }
				random_location_in_market = {
					limit  = { 
						owner ?= root
						can_build_building = building_type:farming_village
						location_unemployed_population_for_building_type = {
							building_type = building_type:farming_village
							value >= 3500 #it's ok if this partially employs, better than nothing
						}
					}
					construct_building = {
						building_type = building_type:farming_village
						cost_multiplier = 0
						cost_multiplier_reason = "ai_broke_so_farm_village_free"
					}
				}
			}
		}
	}
}

#This event looks for markets with large amounts of food and frees up peasants in a random location without a large peasant surplus. This leaves the peasants free to migrate, be employed in other places, etc.
mnt_food.3 = {
	type = country_event
	title = hidden_event.t
	desc = hidden_event.d
	
	trigger = {
		country_exists_and_can_own_locations = yes
		is_ai = yes
	}
	
	hidden = yes
	
	immediate = {
		every_market_present_in_country = {
			limit  = { market_monthly_food_balance > 500 } #much safer check, since there are never positive spikes of food, only negative.
			random_location_in_market = {
				limit = {
					owner ?= root
					has_building_with_at_least_one_level = farming_village
					location_unemployed_population_for_building_type = {
						building_type = building_type:farming_village
						value < 2000
					}
				}
				destroy_building = "building(building_type:farming_village|owner)"
			}
			random_location_in_market = {
				limit = {
					owner ?= root
					has_building_with_at_least_one_level = farming_village
					location_unemployed_population_for_building_type = {
						building_type = building_type:farming_village
						value < 2000
					}
				}
				destroy_building = "building(building_type:farming_village|owner)"
			}
		}	
	}
}
