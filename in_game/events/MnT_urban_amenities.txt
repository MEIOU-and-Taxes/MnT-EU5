namespace = mnt_urban_amenities

#fires once at startup
mnt_urban_amenities.1 = {
    type = country_event
    title = hidden_event.t
    hidden = yes

    immediate = {
		every_owned_location = {
			if = {
				limit = {
					location_rank = location_rank:town
				}
				change_building_level_in_location = {
					building = building_type:urban_amenities
					value = 2
					max = urban_amenities_max_level
				}
			}
			if = {
				limit = {
					location_rank = location_rank:city
				}
				change_building_level_in_location = {
					building = building_type:urban_amenities
					value = 6
					max = urban_amenities_max_level
				}
			}
			if = {
				limit = {
					location_rank = location_rank:megalopolis
				}
				change_building_level_in_location = {
					building = building_type:urban_amenities
					value = 10
					max = urban_amenities_max_level
				}
			}
		} 
    }
}
