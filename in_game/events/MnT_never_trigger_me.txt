# Intentionally untriggered event to satisfy the engine's static variable analyzer.
# Will hide useless errors, doesn't error itself because it's technically triggererd in on_startup (when always = no kek), but the current_year trigger prevents it doing anything just in case. 
# Variables that are not used in code but are still checked for are flagged as "never set"
# We can silence the errors from never set vars by setting the vars in an event that is never triggered. 
# Variables that are set but used only in localization are flagged as "set but never used"
# We can silence the errors from never used vars by multiplying them by 1 in the same never-triggered event, which counts as "used" without changing their value.
# Credits for finding this workaround to HappyMonk3y99, ported and adapted to M&T by Mepper.

namespace = mnt_never_trigger_me

mnt_never_trigger_me.1 = {
	type = country_event
	title = "hidden_event.t"
	desc = "hidden_event.d"
	hidden = yes
	orphan = yes

    trigger = {
        current_year = 42069 #haha
    }

	immediate = {
		change_variable = { name = mnt_country_total_goods_value multiply = 1 }

		# MnT EPBM: routing var set dynamically from the estate map at runtime
		set_global_variable = { name = MnT_epbm_estate_ref value = estate_type:nobles_estate }

		# MnT EPBM: display vars consumed by the estate tooltip
		change_variable = { name = MnT_epbm_show_nobles multiply = 1 }
		change_variable = { name = MnT_epbm_show_clergy multiply = 1 }
		change_variable = { name = MnT_epbm_show_burghers multiply = 1 }
		change_variable = { name = MnT_epbm_show_peasants multiply = 1 }
		change_variable = { name = MnT_epbm_show_dhimmi multiply = 1 }
		change_variable = { name = MnT_epbm_show_tribes multiply = 1 }
		change_variable = { name = MnT_epbm_show_cossacks multiply = 1 }
	}

	option = {
		name = "OK"
	}
}
