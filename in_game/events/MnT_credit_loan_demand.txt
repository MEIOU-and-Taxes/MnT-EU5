namespace = mnt_credit_loan_demand

# Recalculate credit demand from each country's outstanding loans.
mnt_credit_loan_demand.1 = {
	type = country_event
	title = hidden_event.t
	hidden = yes
	trigger = {
		num_loans > 0
	}
	immediate = {
		mnt_accumulate_credit_loan_demand_for_country = yes
	}
}

# Aggregate all capital-market loan demand into temporary market demands.
mnt_credit_loan_demand.2 = {
	type = country_event
	title = hidden_event.t
	hidden = yes
	trigger = {
		tag = FRA
	}
	immediate = {
		mnt_aggregate_credit_loan_demand_markets = yes
	}
}
