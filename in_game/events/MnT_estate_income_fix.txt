namespace = estate_income_fix

# Hidden event to calculate and grant control shortfall payments to estates
# Formula: For each estate, sum across all locations: estate_power * tax_base * ((1 - control)/control)
# This represents the gold that estates should receive from imperfect control over locations
estate_income_fix.1 = {
	type = country_event
	title = estate_income_fix.1.title
	hidden = yes

	trigger = {
		is_ai = no
	}

	# Calculate and grant control shortfall payments to estates
	immediate = {
		mnt_update_control_income_loss_per_estate = { estate = nobles }
		mnt_update_control_income_loss_per_estate = { estate = clergy }
		mnt_update_control_income_loss_per_estate = { estate = burghers }
		mnt_update_control_income_loss_per_estate = { estate = peasants }
		mnt_update_control_income_loss_per_estate = { estate = tribes }
		mnt_update_control_income_loss_per_estate = { estate = dhimmi }
		mnt_update_control_income_loss_per_estate = { estate = cossacks }
	}
}
