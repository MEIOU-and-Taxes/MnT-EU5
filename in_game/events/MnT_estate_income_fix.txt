namespace = estate_income_fix

# Hidden events to calculate and grant control shortfall payments to estates
# Formula: For each estate, sum across all locations: estate_power * tax_base * ((1 - control)/control)
# This represents the gold that estates should receive from imperfect control over locations

# This is fired monthly for the player country to ensure they get the income fix promptly
estate_income_fix.1 = {
	type = country_event
	title = estate_income_fix.1.title
	hidden = yes

	trigger = {
		is_ai = no
	}

	# Calculate and grant control shortfall payments to estates
	immediate = {
		mnt_update_control_income_loss_all_estates = { yearly = 0 }
	}
}

# This is fired yearly for AI countries to ensure they also get the income fix
estate_income_fix.2 = {
	type = country_event
	title = estate_income_fix.2.title
	hidden = yes

	trigger = {
		is_ai = yes
	}
	
	immediate = {
		mnt_update_control_income_loss_all_estates = { yearly = 1 }
	}
}
