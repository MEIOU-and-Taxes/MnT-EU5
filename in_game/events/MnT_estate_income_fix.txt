namespace = estate_income_fix

# Hidden event to calculate and grant control shortfall payments to estates
# Formula: For each estate, sum across all locations: estate_power * tax_base * ((1 - control)/control)
# This represents the gold that estates should receive from imperfect control over locations
estate_income_fix.1 = {
	type = country_event
	title = estate_income_fix.1.title
	hidden = yes

	trigger = {
		is_ai = no
	}

	# Calculate and grant control shortfall payments to estates
	immediate = {
		every_estate = {
			limit = {
				estate_type != estate_type:crown_estate
				power > 0
			}

			prepare_country_estate_variables_for_income_lost = yes

			estate_add_gold = { value = local_var:income_lost }
		}
	}
}

estate_income_fix.2 = {
	type = country_event
	title = estate_income_fix.2.title
	hidden = yes

	trigger = {
		is_ai = yes
	}
	
	immediate = {
		every_estate = {
			limit = {
				estate_type != estate_type:crown_estate
				power > 0
			}

			prepare_country_estate_variables_for_income_lost = yes

			estate_add_gold = {
				value = {
					add = local_var:income_lost
					multiply = 12
				}
			}
		}
	}
}
