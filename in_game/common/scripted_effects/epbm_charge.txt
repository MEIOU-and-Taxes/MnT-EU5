# Copies pay vars to show vars so tooltips display last month's values.
epbm_snapshot_for_display = {
	if = {
		limit = { has_variable = epbm_show_crown_pool }
		set_variable = { name = epbm_prev_crown_pool value = var:epbm_show_crown_pool }
	}

	if = {
		limit = { has_variable = epbm_show_crown_bldg }
		set_variable = { name = epbm_prev_crown_bldg value = var:epbm_show_crown_bldg }
	}

	epbm_for_each_estate = { effect = epbm_snapshot_estate }

	if = {
		limit = { has_variable = epbm_crown_pool_cost }
		set_variable = { name = epbm_show_crown_pool value = var:epbm_crown_pool_cost }
	}

	if = {
		limit = { has_variable = epbm_crown_bldg_cost }
		set_variable = { name = epbm_show_crown_bldg value = var:epbm_crown_bldg_cost }
	}

	epbm_for_each_estate = { effect = epbm_snapshot_mod }
}
