# Calculate the proportion of power from this estate in the location
mnt_location_pop_power_ratio_per_estate = {
	if = {
		limit = { owner = { country_has_estate = estate_type:$estate$_estate } }
		owner.estate_type:$estate$_estate = { save_temporary_scope_as = target_estate }

		if = {
			limit = { NOT = { estate_type:$estate$_estate = estate_type:peasants_estate } }
			set_local_variable = {
				name = mnt_location_estate_power_$estate$
				value = "local_estate_power(scope:target_estate)"
			}
		}
		else = {
			set_local_variable = {
				name = mnt_location_estate_power_$estate$
				value = {
					value = "local_estate_power(scope:target_estate)"
					multiply = {
						value = num_pop_type:$pop$
						divide = {
							value = num_pop_type:peasants
							add = num_pop_type:laborers
							add = num_pop_type:soldiers
						}
					}
				}
			}
		}

		set_local_variable = {
			name = mnt_location_estate_power_ratio_$pop$
			value = {
				value = local_var:mnt_location_estate_power_$estate$
				divide = {
					value = owner.var:mnt_estate_total_local_estate_power_$estate$
					min = 0.1 # DBZ prevention
				}
			}
		}
	}
}

# Calculate the proportion of population from this estate in the location
mnt_location_pop_ratio_per_estate = {
	if = {
		limit = { owner = { country_has_estate = estate_type:$estate$_estate } }

		set_local_variable = {
			name = mnt_location_estate_pop_ratio_$pop$
			value = {
				value = num_pop_type:$pop$
				divide = {
					value = owner.var:mnt_estate_total_pops_$estate$
					min = 0.1 # DBZ prevention
				}
			}
		}
	}
}

mnt_location_pop_income_distribution_per_estate = {
	if = {
		limit = { owner = { country_has_estate = estate_type:$estate$_estate } }
		# Calc location pop wealth income from expected tax base income distributed by local power
		set_local_variable = {
			name = mnt_location_pop_wealth_income_$estate$
			value = {
				value = owner.var:mnt_estate_wealth_expected_income_$estate$
				multiply = local_var:mnt_location_estate_power_ratio_$pop$
			}
		}
		# Calc location pop wealth income from untracked changes to estate wealth distributed by local population proportion
		set_local_variable = {
			name = mnt_location_pop_wealth_untracked_change_income_$estate$
			value = {
				value = owner.var:mnt_estate_wealth_untracked_change_$estate$
				multiply = local_var:mnt_location_estate_pop_ratio_$pop$
			}
		}
		# Update Location Pop Wealth Pool
		set_variable = {
			name = mnt_location_pop_wealth_$pop$
			value = {
				value = var:mnt_location_pop_wealth_$pop$
				add = local_var:mnt_location_pop_wealth_income_$estate$
				add = local_var:mnt_location_pop_wealth_untracked_change_income_$estate$
			}
		}
		change_variable = {
			name = mnt_location_pop_wealth_total
			add = var:mnt_location_pop_wealth_$pop$
		}
	}
}

### Estate Wealth Pool Tracking:
### --- Variables ---
# mnt_estate_wealth_$estate$						  | tracks the current wealth of an estate
# mnt_estate_prev_wealth_$estate$					  | tracks the previous calculation iteration wealth of the estate
# mnt_estate_wealth_income_lost_from_control_$estate$ | tracks the expected income lost due to lack of control
# mnt_estate_wealth_expected_income_$estate$		  | tracks the expected income based on taxable income - Taxes Paid + expected income lost from control
# mnt_estate_wealth_untracked_change_$estate$		  | tracks changes to estate wealth not explained by expected income (events, etc)
mnt_init_estate_wealth_variables = {

		limit = { NOT = { has_variable = mnt_estate_wealth_vars_setup } } 

		set_variable = { name = mnt_estate_wealth_vars_setup value = 1 }

		mnt_init_estate_wealth_variables_per_estate = { estate = nobles }
		mnt_init_estate_wealth_variables_per_estate = { estate = clergy }
		mnt_init_estate_wealth_variables_per_estate = { estate = burghers }
		mnt_init_estate_wealth_variables_per_estate = { estate = peasants }
		mnt_init_estate_wealth_variables_per_estate = { estate = tribes }
		mnt_init_estate_wealth_variables_per_estate = { estate = dhimmi }
		mnt_init_estate_wealth_variables_per_estate = { estate = cossacks }
	}
}

			set_variable = {

				value = var:mnt_estate_wealth_$estate$
			}	
		}

		estate_type:$estate$_estate = { save_temporary_scope_as = target_estate }

			value = "estate(scope:target_estate).estate_gold"

# Calc expected estate income
mnt_update_estate_wealth_income_per_estate = {
	if = {
		limit = { country_has_estate = estate_type:$estate$_estate }

		estate_type:$estate$_estate = { save_temporary_scope_as = target_estate }

		every_owned_location = {
			if = {
				limit = { local_control > 0 }
				set_variable = {
					name = mnt_estate_wealth_income_lost_from_control_$estate$
					value = {
						value = location_tax_base
						divide = local_control
						subtract = location_tax_base
						multiply = mnt_estate_control_loss_wealth_local_wastage_ratio #for the locals and wastage
						multiply = "local_relative_estate_power(scope:target_estate)"
						min = 0
					}
				}
			}
			else = {
				calc_location_total_profit_value_added = yes
				set_variable = {
					name = mnt_estate_wealth_income_lost_from_control_$estate$
					value = {
						value = local_var:mnt_loc_total_value_added
						multiply = mnt_estate_control_loss_wealth_local_wastage_ratio #for the locals and wastage
						multiply = "local_relative_estate_power(scope:target_estate)"
						min = 0
					}
				}
			}
		}

		random_estate = {
			limit = { estate_type = estate_type:$estate$_estate }
			estate_add_gold = { value = prev.var:mnt_estate_wealth_income_lost_from_control_$estate$ }
		}

		set_variable = {
			name = mnt_estate_wealth_expected_income_$estate$
			value = {
				value = "estate(scope:target_estate).estate_taxable_income"
				subtract = "estate(scope:target_estate).estate_tax" # Remove taxed income that goes to the state
				add = var:mnt_estate_wealth_income_lost_from_control_$estate$
			}
		}
	}
}

# Calc total income unrelated changes to estate wealth since last calculation
mnt_update_estate_wealth_untracked_income_per_estate = {

	if = { 
		limit = { has_variable = mnt_estate_prev_wealth_$estate$ } # Only calculate if we have a previous value to compare against

		set_variable = {
			name = mnt_estate_wealth_untracked_change_$estate$
			value = {
				value = var:mnt_estate_wealth_$estate$
				subtract = var:mnt_estate_prev_wealth_$estate$
				subtract = var:mnt_estate_wealth_expected_income_$estate$
			}
		}
	}
}

# Calculate total population of an estate across the country
# Sum the location pop-income proportion over ALL owned locations
mnt_country_pop_income_proportion_sum = {
	# Total Power Variables
	set_variable = { name = mnt_estate_total_local_estate_power_nobles 	 value = 0 }
	set_variable = { name = mnt_estate_total_local_estate_power_clergy 	 value = 0 }
	set_variable = { name = mnt_estate_total_local_estate_power_burghers value = 0 }
	set_variable = { name = mnt_estate_total_local_estate_power_tribes 	 value = 0 }
	set_variable = { name = mnt_estate_total_local_estate_power_peasants value = 0 }

	# Total Power Variables
	set_variable = { name = mnt_estate_total_pops_nobles   value = num_pop_type_in_country:nobles }
	set_variable = { name = mnt_estate_total_pops_clergy   value = num_pop_type_in_country:clergy }
	set_variable = { name = mnt_estate_total_pops_burghers value = num_pop_type_in_country:burghers }
	set_variable = { name = mnt_estate_total_pops_tribes   value = num_pop_type_in_country:tribesmen }
	set_variable = { 
		name = mnt_estate_total_pops_peasants 
		value = {
			value = num_pop_type_in_country:peasants
			add = num_pop_type_in_country:laborers
			add = num_pop_type_in_country:soldiers
		}
	}

	# iterate all owned locations
	every_owned_location = {
		mnt_location_local_estate_power_per_estate = { estate = nobles 	 }
		mnt_location_local_estate_power_per_estate = { estate = clergy 	 }
		mnt_location_local_estate_power_per_estate = { estate = burghers }
		mnt_location_local_estate_power_per_estate = { estate = tribes 	 }
		mnt_location_local_estate_power_per_estate = { estate = peasants }
		mnt_location_local_estate_power_per_estate = { estate = peasants }
		mnt_location_local_estate_power_per_estate = { estate = peasants }

		prev = {
			# add per-location value into country accumulator
			change_variable = { name = mnt_estate_total_local_estate_power_nobles	 add = prev.var:mnt_location_estate_local_power_nobles }
			change_variable = { name = mnt_estate_total_local_estate_power_clergy 	 add = prev.var:mnt_location_estate_local_power_clergy }
			change_variable = { name = mnt_estate_total_local_estate_power_burghers  add = prev.var:mnt_location_estate_local_power_burghers }
			change_variable = { name = mnt_estate_total_local_estate_power_tribes 	 add = prev.var:mnt_location_estate_local_power_tribes }
			change_variable = { name = mnt_estate_total_local_estate_power_peasants  add = prev.var:mnt_location_estate_local_power_peasants }
		}
	}
}
