remove_RGOs = {
	error_log = "::START:RGO removal"

	every_owned_location = {
		replace_rgos_with_buildings = yes
		if = {
			limit = { raw_material = goods:wheat }
			change_building_level_in_location = {
				building = building_type:RGO_building_wheat
				value = raw_material_output
			}
		}
	}

	every_owned_location = {
		change_max_raw_material_workers = -100
	}

	add_country_modifier = {
		modifier = destroy_RGO
		years = -1
		mode = add
		desc = "Sample text"
	}

	error_log = "::END  :RGO removal"
}
