calc_location_total_profit_value_added = {
    save_scope_as = target_location
    # sum building profits in this location
    set_local_variable = { name = mnt_loc_building_profit value = 0 }
    every_buildings_in_location = {
        # building scope here
        scope:target_location = {
            change_local_variable = {
                name = mnt_loc_building_profit
                add = prev.building_profit
            }
        }
    }
    # add RGO value
    set_local_variable = {
        name = mnt_loc_rgo_value
        value = {
            value = raw_material_output
            multiply = "scope:target_location.raw_material.price_in_market(scope:target_location.market)"
        }
    }
    # calc total 
    set_local_variable = {
        name = mnt_loc_total_value_added
        value = {
            value = local_var:mnt_loc_building_profit
            add = local_var:mnt_loc_rgo_value
        }
    }
}
