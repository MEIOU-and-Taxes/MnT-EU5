run_logging = {
	if = {
		limit = { has_global_variable = is_logging_enabled}
		print_regional_building_types = yes        # characters at start: 147,474
		print_regional_population = yes            # characters at start:   9,505
		print_regional_prices = yes                # characters at start: 765,974
		print_regional_roads = yes
		print_market_information = yes
		print_country_information = yes
		error_log = "::Done logging for this year"
	}
}

print_regional_population = {
    every_region = {
		error_log = "::POP::[GetCurrentYear]:[THIS.GetRegion.GetNameWithNoTooltip]:[THIS.GetRegion.GetTotalPopulation]"
    }
}

print_regional_prices = {
	every_goods = {
		# limit = { this = goods:wheat }
		save_scope_as = good_curr
		every_region = {
			set_local_variable = {
				name = avg_cost
				value = {
					every_location_in_region = {
						limit = { is_land = yes }
						add = "market.market_price(scope:good_curr)"
					}
					divide = location_counter
				}

			}
			if = {
				limit = { location_counter > 0 }
				error_log = "::GP::[GetCurrentYear]:[PREV.GetGoods.GetNameWithNoTooltip]:[THIS.GetRegion.GetNameWithNoTooltip]:[SCOPE.GetLocalVariable('avg_cost').GetValue]"
			}
		}
	}
	remove_local_variable = avg_cost
}

print_regional_building_types = {
    every_building_type = {
        set_local_variable = { name = bldg_curr value = this }
        every_region = {
            set_local_variable = {
                name = bldg_count
                value = {
                    every_location_in_region = {
                        every_buildings_in_location = {
                            limit = { building_type = local_var:bldg_curr }
                            add = building_level
                        }

                    }
                }
            }

            if = {
                limit = { local_var:bldg_count > 0 }
				error_log = "::BT:[GetCurrentYear]:[SCOPE.GetLocalVariable('bldg_curr').GetBuildingType.GetNameWithNoTooltip]:[SCOPE.GetLocalVariable('bldg_count').GetValue]:[THIS.GetRegion.GetNameWithNoTooltip]"
            }
        }
    }
	remove_local_variable = bldg_count
}

print_regional_roads = {
	every_road_type = {
		if = {
			limit = {
				OR = {
					this = road_type:gravel_road
					AND = {
						this = road_type:paved_road
						current_age_or_later = { age = age_3_discovery }
					}
					AND = {
						this = road_type:modern_road
						current_age_or_later = { age = age_5_absolutism }
					}
					AND = {
						this = road_type:railroad
						current_age_or_later = { age = age_6_revolutions }
					}
				}
			}

			set_local_variable = { name = road_curr value = this }
			every_region = {
				set_local_variable = { name = num_locations value = 0 }
				set_local_variable = { name = num_locations_with_road value = 0 }
				every_ownable_location_in_region = {
					change_local_variable = { name = num_locations add = 1 }

					if = {
						limit = { has_local_variable = loc_has_road }
						remove_local_variable = loc_has_road
					}

					every_neighbor_location = {
						if = {
							limit = {
								has_road_of_type_to = {
									target = prev
									type = local_var:road_curr
								}
							}
							set_local_variable = loc_has_road
						}
					}
					if = {
						limit = { has_local_variable = loc_has_road }
						change_local_variable = { name = num_locations_with_road add = 1 }
					}
				}
				if = {
					limit = { local_var:num_locations_with_road > 0 }
					error_log = "::RT::[GetCurrentYear]:[THIS.GetRegion.GetNameWithNoTooltip]:[SCOPE.GetLocalVariable('road_curr').GetRoadType.GetNameWithNoTooltip]:[SCOPE.GetLocalVariable('num_locations').GetValue]:[SCOPE.GetLocalVariable('num_locations_with_road').GetValue]"
				}
			}
		}
	}

	# error_log = "[SCOPE.GetLocalVariable('connections').GetValue]"

	remove_local_variable = road_curr
	remove_local_variable = num_locations
	remove_local_variable = num_locations_with_road
}
