mnt_update_control_income_loss_per_estate = {
	estate_type:$estate$_estate = { save_temporary_scope_as = target_estate }

	set_variable = { name = mnt_estate_wealth_income_lost_from_control_$estate$ value = 0 }
	every_owned_location = {
		if = {
			limit = { local_control > 0 }
			set_variable = {
				name = mnt_estate_wealth_income_lost_from_control_$estate$
				value = {
					value = location_tax_base
					divide = local_control
					subtract = location_tax_base
					multiply = 0.95 #for the locals and wastage
					multiply = "local_relative_estate_power(scope:target_estate)"
					min = 0
				}
			}
			prev = { # PREV scope is the country, value of all locations is summed up
				change_variable = { 
					name = mnt_estate_wealth_income_lost_from_control_$estate$ 
					add = prev.var:mnt_estate_wealth_income_lost_from_control_$estate$ # here prev scope is the location
				}
			}
		}
		else = {
			calc_location_total_profit_value_added = yes
			set_variable = {
				name = mnt_estate_wealth_income_lost_from_control_$estate$
				value = {
					value = local_var:mnt_loc_total_value_added
					multiply = 0.95 #for the locals and wastage
					multiply = "local_relative_estate_power(scope:target_estate)"
					min = 0
				}
			}
			prev = { # PREV scope is the country, value of all locations is summed up
				change_variable = { 
					name = mnt_estate_wealth_income_lost_from_control_$estate$ 
					add = prev.var:mnt_estate_wealth_income_lost_from_control_$estate$ # here prev scope is the location
				}
			}
		}
	}

	random_estate = {
		limit = { estate_type = estate_type:$estate$_estate }
		estate_add_gold = { value = prev.var:mnt_estate_wealth_income_lost_from_control_$estate$ }
	}
}
