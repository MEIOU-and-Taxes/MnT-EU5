# Calculate heuristic scores for centers.
mnt_calculate_center_scores = {
	every_location_in_the_world = {
		# Trade heuristic: weighted blend of access, local advantage, burgher capacity, and state trade capacity from buildings.
		set_variable = {
			name = mnt_trade_score
			value = {
				value = market_access
				multiply = 5
				add = {
					value = modifier:local_merchant_power
					multiply = 5
				}
				add = {
					value = modifier:local_trades_per_burgher
					multiply = num_pop_type:burghers
					multiply = 20
				}
				add = {
					value = modifier:merchant_capacity_from_building
					multiply = 10
				}
				add = {
					value = location_maritime_merchant_power
					multiply = 5
				}
				add = {
					value = is_market_center
					multiply = 25 #boolean * 20
				}
			}
		}

		# Production heuristic: Total goods produced in location / population
		set_variable = {
			name = mnt_production_score
			value = {
				value = goods_output
				divide = {
					value = population
				}
			}
		}

		# Culture heuristic: cultural influence + tradition
		set_variable = {
			name = mnt_culture_score
			value = {
				value = modifier:local_cultural_influence
				multiply = 33
				add = {
					value = modifier:local_cultural_tradition
					multiply = 10
				}
				add = {
					value = location_art_quality
					multiply = 12.5
				}
			}
		}

		# Education heuristic: clergy + nobles + literacy + buildings.
		set_variable = {
			name = mnt_education_score
			value = {
				value = num_pop_type:clergy
				multiply = 0.5
				add = {
					value = num_pop_type:nobles
					multiply = 0.25
				}
				add = {
					value = average_location_literacy
					multiply = 20
				}
				if = {
					limit = { has_building = university }
					add = 50
				}
				if = {
					limit = { has_building = monastery }
					add = 25
				}
				if = {
					limit = { has_building = library }
					add = 30
				}
			}
		}
	}
}
