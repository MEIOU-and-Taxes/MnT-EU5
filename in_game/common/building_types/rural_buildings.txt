			text = local_bailiff_tt
			scope:actor = {	total_building_levels_including_construction:bailiff < modifier:num_bailiffs }
		}
		owner = scope:actor #not in subjects
	}
	
	
	unique_production_methods = {
		bailiff_input = {
			paper = 0.1
			tools = 0.15
			weaponry = 0.25
			stone = 0.1
			category = building_maintenance
		}
	}

	modifier = {
		local_max_control = 0.25 #M&T from 0.2
		local_nobles_estate_power = 1.0
		local_peasant_enfranchisment = -0.1
	}
	
	
	on_built = {
		location = { add_to_variable_list = { name = MnT_epbm_buildings target = prev } }
		MnT_epbm_on_building_built = yes
	}
	on_destroyed = {
		location = { remove_list_variable = { name = MnT_epbm_buildings target = prev } }
		MnT_epbm_on_building_destroyed = yes
	}
}
