# MnT EPBM: on_built / on_destroyed hooks in this file are maintained by
# tools/MnT_epbm_generator/. After adding a building or a maintenance
# production method, commit your changes and re-run:

mnt_hundi_house = {
	is_foreign = no
	max_levels = 1
	pop_type = burghers
	important_for_AI = yes

	category = trade_category
	employment_size = generic_burgher_employment

	rural_settlement = no
	town = yes
	city = yes
	megalopolis = yes
	build_time = generic_town_build_time

	country_potential = {
		OR = {
			culture = { has_culture_group = culture_group:indian_group }
			original_capital ?= { sub_continent = sub_continent:south_asia }
		}
	}

	possible_production_methods = {
		mnt_hundi_house_maintenance
	}

	construction_demand = town_building_construction
	custom_tags = { guild }
	on_built = {
		location = { add_to_variable_list = { name = MnT_epbm_buildings target = prev } }
		MnT_epbm_on_building_built = yes
	}
	on_destroyed = {
		location = { remove_list_variable = { name = MnT_epbm_buildings target = prev } }
		MnT_epbm_on_building_destroyed = yes
	}
}
