on_game_start_MnT = {
	effect = {
		# Initialize center of importance systems on game start
		random_country = { trigger_event_silently = { id = mnt_centers.1 } }
		every_market_in_world = {	#Helicopter-tools to make the world go vroom
			add_goods_supply = {
				goods = goods:tools
				amount = 250
			}
		}

		# Initialize GDP update frequency variable and trigger first update
		every_country = {
			trigger_event_silently = { id = rgo_removal.01 }
			if = {
				limit = { is_ai = no }
				if = {
					limit = { NOT = { has_variable = mnt_goods_value_update_freq } }
					set_variable = { name = mnt_goods_value_update_freq value = 1 }
				}
			}
			trigger_event_silently = { id = mnt_goods_value.1 }
		}
		# Handle Pop and Food building startup
		every_country = { limit = { num_locations > 0 }	trigger_event_silently = { id = mnt_food.1 } }
	}
}

pulse_yearly_MnT = {
	trigger = {
		great_power_ranking = 1
	}
	effect = {
		random_country = {
			trigger_event_silently = { id = mnt_centers.1 }
			trigger_event_silently = { id = LOGGER_CHARTS.02 }
		}
	}
}

pulse_yearly_per_country_MnT = {
	events = {
		mnt_food.2
		mnt_goods_value.2
	}
}

biyearly_country_pulse_MnT = {
	events = {
		mnt_food.3
	}
}

monthly_country_pulse_MnT = {
	events = {
		mnt_goods_value.1

		# Calculate demands of individual countries
		cost_of_the_court.1
		# Ensure that the calculation results are available
		# Calculate the demands and estate transfer of individual countries
		diplomatic_spending.1
		# Ensure that the calculation results are available

		delay = { days = 1 }

		# Aggregate the total CotC demands in each market
		cost_of_the_court.2
		# Aggregate the total diplomatic spending demands in each market
		diplomatic_spending.2
	}
}
