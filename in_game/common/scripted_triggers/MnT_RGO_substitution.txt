can_build_clay_RGO = {
	OR = {
		raw_material = goods:clay
		is_coastal = yes
		has_river = yes
		is_adjacent_to_lake = yes
	}
}

can_build_fruit_RGO = {
	OR = {
		raw_material = goods:fruit
		vegetation = grasslands
		vegetation = farmland
		vegetation = woods
		vegetation = jungle
	}
}
can_build_lumber_RGO = {
	OR = {
		raw_material = goods:lumber
		vegetation = woods
		vegetation = forest
		vegetation = farmland
		vegetation = jungle
	}
}
