MnT_can_have_nomadic_wool_pasture = {
	OR = {
		sub_continent = sub_continent:south_america
		continent = continent:africa # Sheep had long proliferated throughout the entire African continent
		continent = continent:europe
		AND = {
			continent = continent:asia
			NOR = {
				# Sheep were not herded in Japan nor Indonesia, but were in the rest of Asia
				region = region:japan_region
				region = region:indonesia_region
			}
		}
	}
	NOT = {
		is_in_scripted_geography = scripted_geography:MnT_tsetse_geography # Can't have pastures in sleeping sickness areas
	}
}

REPLACE:city_illustration_trigger = {
	OR = {
		location_rank ?= location_rank:city
		location_rank ?= location_rank:megalopolis
	}
}

REPLACE:town_illustration_trigger = {
	is_city = yes # actually means is urban, so includes cities and towns
	NOR = {
		location_rank ?= location_rank:city
		location_rank ?= location_rank:megalopolis
	}
}
