### M&T replaces all the Vanilla climates with Koppen climates
### In this file I'm creating some scripted_triggers, which group together some of our climates
### Mostly used to replace Vanilla triggers
### Find M&T climates in: in_game/common/climates/MnT_default.txt
### Created initially by Micha 'Mepper' Helder in May 2026

### 8 Vanilla Climate triggers to group Koppen climates and bind them all into darkness
### Vanilla has a trigger for climate? Replace with one of these to continue non-sensible vanilla climate assignment
### I assigned sensible things to non-sensible things, so no way on earth that this makes sense
arctic_climate_trigger = {
    OR = {
        climate = mediterranean_boreal_climate
        climate = mediterranean_hypercontinental_climate
        climate = monsoon_boreal_climate
        climate = monsoon_hypercontinental_climate
        climate = boreal_climate
        climate = hypercontinental_climate
        climate = tundra_climate
        climate = polar_climate
    }
}

tropical_climate_trigger = {
    OR = {
        climate = rainforest_climate
        climate = monsoon_climate
        climate = savanna_summer_climate
        climate = savanna_winter_climate
    }
}

continental_climate_trigger = {
    OR = {
        climate = continental_climate
        climate = hemiboreal_climate
        climate = monsoon_continental_climate
        climate = monsoon_hemiboreal_climate
        climate = andean_tundra_climate
        climate = mediterranean_hemiboreal_climate
    }
}

arid_climate_trigger = {
    OR = {
        climate = hot_desert_climate
        climate = hot_steppe_climate
    }
}

cold_arid_climate_trigger = {
    OR = {
        climate = cold_desert_climate
        climate = cold_steppe_climate
    }
}

mediterranean_climate_trigger = {
    OR = {
        climate = mediterranean_climate
        climate = oceanic_mediterranean_climate
        climate = subalpine_mediterranean_climate
        climate = mediterranean_continental_climate
    }
}

subtropic_climate_trigger = {
    OR = {
        climate = humid_subtropical_climate
        climate = monsoon_humid_subtropical_climate
        climate = monsoon_oceanic_climate
        climate = monsoon_subpolar_oceanic_climate
    }
}

oceanic_climate_trigger = {
    OR = {
        climate = oceanic_climate
        climate = subpolar_oceanic_climate
    }
}
