is_road_available = {
	OR = {
		this = road_type:gravel_road
		AND = {
			this = road_type:paved_road
			current_age_or_later = { age = age_3_discovery }
		}
		AND = {
			this = road_type:modern_road
			current_age_or_later = { age = age_5_absolutism }
		}
		AND = {
			this = road_type:railroad
			current_age_or_later = { age = age_6_revolutions }
		}
	}
}
