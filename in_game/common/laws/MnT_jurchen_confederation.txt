REPLACE:economic_framework = {
	type = international_organization
	law_category = socioeconomic
	requires_vote = no
	potential = {
		international_organization_type = international_organization_type:jurchen_confederation
	}
	allow = {
		leader_country ?= { tribal_cohesion > 10 }
	}
	locked = {
		custom_tooltip = {
			text = 5_years_cd
			leader_country = { has_variable = economic_framework_cooldown_variable }
		}
	}

	shared_crops_policy = {
		wants_this_policy_bias = {
			add = {
				desc = "POLICY_BIAS_GOOD_IDEA"
				value = 20
			}
		}

		on_activate = {
			leader_country = { 
				set_variable = {
					name = economic_framework_cooldown_variable
					value = yes
					years = 5
				}
				add_tribal_cohesion = tribal_cohesion_weak_penalty
				if = {
					limit = {
						ruler_or_regent ?= { has_variable = difficult_sinicization_variable }
					}
					add_tribal_cohesion = tribal_cohesion_mild_penalty
					add_stability = stability_mild_penalty
				}
			}
		}
		years = 5
		country_modifier = {
			monthly_towards_traditional_economy = societal_value_monthly_move
			#MnT removed: global_monthly_food_modifier = 0.1
			monthly_towards_unsinicized = societal_value_monthly_move
			monthly_tribal_cohesion = 0.03
		}
	}

	decentralized_farming_villages = {
		wants_this_policy_bias = {
			add = {
				desc = "POLICY_BIAS_GOOD_IDEA"
				value = 20
			}
		}

		on_activate = {
			leader_country = { 
				set_variable = {
					name = economic_framework_cooldown_variable
					value = yes
					years = 5
				}
				add_tribal_cohesion = tribal_cohesion_weak_penalty
				if = {
					limit = {
						ruler_or_regent ?= { has_variable = difficult_sinicization_variable }
					}
					add_tribal_cohesion = tribal_cohesion_mild_penalty
					add_stability = stability_mild_penalty
				}
			}
		}
		years = 5
		country_modifier = {
			monthly_towards_traditional_economy = societal_value_minor_monthly_move
			monthly_towards_sinicized = societal_value_monthly_move
			stability_investment = small_stability_investment
		}
	}

	financed_rgo_policy = {
		wants_this_policy_bias = {
			add = {
				desc = "POLICY_BIAS_GOOD_IDEA"
				value = 20
			}
		}

		on_activate = {
			leader_country = { 
				set_variable = {
					name = economic_framework_cooldown_variable
					value = yes
					years = 5
				}
				add_tribal_cohesion = tribal_cohesion_weak_penalty
				if = {
					limit = {
						ruler_or_regent ?= { has_variable = difficult_sinicization_variable }
					}
					add_tribal_cohesion = tribal_cohesion_mild_penalty
					add_stability = stability_mild_penalty
				}
			}
		}
		years = 5
		country_modifier = {
			global_max_rgo_size_modifier = 0.2
			monthly_towards_capital_economy = societal_value_monthly_move
			monthly_towards_sinicized = societal_value_monthly_move
			monthly_tribal_cohesion = -0.05
		}
	}
}
