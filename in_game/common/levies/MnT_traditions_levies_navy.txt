#the way naval levies are assessed is a bit different from land levies. they are still evaluated on a by-pop basis, and you still get 1 boat per pop
#the "crew" are I assume the actual people who die if the boat is destroyed
#if your crew and the evaluated pop are the same type, then the levy size will constrain the total number of ships which may be spawned to how many of the unit could be manned by the levied number of pops
#otherwise its constrained to the total size of the allow_as_crew type
#as far as i can tell, you cannot use the same pop type across multiple crews or multiple levies
#you don't appear to be able to use burghers as crewmen and the evaluated pop type at the same time for whatever reason, the levy just won't spawn in the game- this isn't true of other pop types though
#M&T added unit
levy_eastindiaman = {
	size = 0.001
	
	allow = {
		pop_type = pop_type:burghers
		location ?= {
			has_building_with_at_least_one_level = wharf
		}
	}

	allow_as_crew = {
		OR = {
			pop_type = pop_type:laborers
			#pop_type = pop_type:burghers
			pop_type = pop_type:peasants
		}
	}

	unit = n_eastindiaman_levy
}
