#the way naval levies are assessed is a bit different from land levies. they are still evaluated on a by-pop basis, you get 1 boat per pop
#the "crew" are I assume the actual people who die if the boat is destroyed
#if your crew and the evaluated pop are the same type, then the levy size will constrain the total number of ships which may be spawned to how many of the unit could be manned by the levied number of pops
#otherwise its constrained to the total size of the allow_as_crew type
#as far as i can tell, you cannot use the same pop type across multiple crews or multiple levies
#you don't appear to be able to use burghers as crewmen for whatever reason, the levy just won't spawn in the game

#MnT replaces burghers and peasants in fishing boat levies with laborers
REPLACE:levy_fishing_boat = {
	size = 0.01
	
	allow = {
		location ?= {
			raw_material = goods:fish
		}
		pop_type = pop_type:laborers
	}

	#crew members. Number per boat is set by the sub unit definition, in this case n_fishing_boat
	allow_as_crew = {
		pop_type = pop_type:laborers
	}

	unit = n_fishing_boat
}
