mnt_income_graph_update_freq_toggle = {
	scope = country
	is_shown = { is_ai = no }
	is_valid = { is_ai = no }

	effect = {
		if = {
			limit = { NOT = { has_variable = mnt_income_graph_update_freq } }
			set_variable = { name = mnt_income_graph_update_freq value = 0 }
		}

		if = {
			limit = { var:mnt_income_graph_update_freq = 0 }
			set_variable = { name = mnt_income_graph_update_freq value = 1 }
		}
		else = {
			set_variable = { name = mnt_income_graph_update_freq value = 0 }
		}

		mnt_income_graph_refresh_display_from_mode = yes
	}
}

mnt_income_graph_metric_toggle = {
	scope = country
	is_shown = { is_ai = no }
	is_valid = { is_ai = no }

	effect = {
		if = {
			limit = { NOT = { has_variable = mnt_income_graph_metric_mode } }
			set_variable = { name = mnt_income_graph_metric_mode value = 0 }
		}

		if = {
			limit = { var:mnt_income_graph_metric_mode = 0 }
			set_variable = { name = mnt_income_graph_metric_mode value = 1 }
		}
		else_if = {
			limit = { var:mnt_income_graph_metric_mode = 1 }
			set_variable = { name = mnt_income_graph_metric_mode value = 2 }
		}
		else = {
			set_variable = { name = mnt_income_graph_metric_mode value = 0 }
		}

		mnt_income_graph_refresh_display_from_mode = yes
	}
}
