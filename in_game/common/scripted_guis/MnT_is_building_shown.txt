mnt_is_building_shown = {
	scope = building_type
	saved_scopes = {
		player
	}
	is_shown = {
		OR = {
			AND = {
				scope:player = { has_variable = mnt_force_show_rgo_category }
				is_target_in_global_variable_list = {
					name = mnt_global_hidden_buildings
					target = this
				}
			}
			AND = {
				scope:player = { has_variable = mnt_force_show_non_rgo_category }
				NOT = {
					is_target_in_global_variable_list = {
						name = mnt_global_hidden_buildings
						target = this
					}
				}
			}
			AND = {
				scope:player = { NOT = { has_variable = mnt_force_show_rgo_category } }
				scope:player = { NOT = { has_variable = mnt_force_show_non_rgo_category } }
				OR = {
					scope:player = {
						is_target_in_variable_list = {
							name = mnt_country_force_show_buildings
							target = prev
						}
					}
					AND = {
						scope:player = { has_variable = mnt_show_rgo_buildings }
						is_target_in_global_variable_list = {
							name = mnt_global_hidden_buildings
							target = this
						}
					}
					NOT = {
						OR = {
							is_target_in_global_variable_list = {
								name = mnt_global_hidden_buildings
								target = this
							}
							scope:player = {
								is_target_in_variable_list = {
									name = mnt_country_hidden_buildings
									target = prev
								}
							}
						}
					}
				}
			}
		}
	}
}
mnt_is_building_hidden = {
	scope = building_type
	saved_scopes = {
		player
	}
	is_shown = {
		OR = {
			is_target_in_global_variable_list = {
				name = mnt_global_hidden_buildings
				target = this
			}
			scope:player = {
				is_target_in_variable_list = {
					name = mnt_country_hidden_buildings
					target = prev
				}
			}
		}
	}
}
