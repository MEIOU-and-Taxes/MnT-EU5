mnt_toggle_rgo_buildings_visibility = {
	scope = country
	is_shown = { is_ai = no }
	is_valid = { is_ai = no }

	effect = {
		if = {
			limit = { has_variable = mnt_show_rgo_buildings }
			remove_variable = mnt_show_rgo_buildings
		}
		else = {
			set_variable = { name = mnt_show_rgo_buildings value = 1 }
		}
	}
}

mnt_is_rgo_buildings_visible = {
	scope = country
	is_shown = { has_variable = mnt_show_rgo_buildings }
}
