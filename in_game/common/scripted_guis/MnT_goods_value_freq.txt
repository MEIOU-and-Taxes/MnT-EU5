mnt_goods_value_update_freq_cycle = {
	scope = country
	is_shown = { is_ai = no }
	is_valid = { is_ai = no }

	effect = {
		if = {
			limit = { NOT = { has_variable = mnt_goods_value_update_freq } }
			set_variable = { name = mnt_goods_value_update_freq value = 1 }
		}

		if = {
			limit = { var:mnt_goods_value_update_freq = 0 }
			set_variable = { name = mnt_goods_value_update_freq value = 1 }
		}
		else_if = {
			limit = { var:mnt_goods_value_update_freq = 1 }
			set_variable = { name = mnt_goods_value_update_freq value = 2 }
		}
		else = {
			set_variable = { name = mnt_goods_value_update_freq value = 0 }
		}

		if = {
			limit = { var:mnt_goods_value_update_freq <= 1 }
			calc_country_goods_value_produced = yes
		}
	}
}
