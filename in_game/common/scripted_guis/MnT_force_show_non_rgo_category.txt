mnt_force_show_non_rgo_category_on = {
	scope = country
	is_shown = { is_ai = no }
	is_valid = { is_ai = no }

	effect = {
		remove_variable = mnt_force_show_rgo_category
		set_variable = { name = mnt_force_show_non_rgo_category value = 1 }
	}
}

mnt_force_show_non_rgo_category_off = {
	scope = country
	is_shown = { is_ai = no }
	is_valid = { is_ai = no }

	effect = {
		remove_variable = mnt_force_show_non_rgo_category
		remove_variable = mnt_country_force_show_buildings
	}
}
