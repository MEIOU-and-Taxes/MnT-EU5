# Urban/rural goods value helpers for locations. These rely on the variables
# populated by calc_urban_goods_value_produced_location /
# calc_rural_goods_value_produced_location.
mnt_location_urban_goods_value = {
	if = {
		limit = { has_variable = mnt_loc_building_profit }
		add = { value = var:mnt_loc_building_profit }
	}
}

mnt_location_rural_goods_value = {
	if = {
		limit = { has_variable = mnt_rural_goods_value }
		add = { value = var:mnt_rural_goods_value }
	}
}

mnt_location_total_goods_value = {
	if = {
		limit = { has_variable = mnt_loc_building_profit }
		add = { value = var:mnt_loc_building_profit }
	}
	if = {
		limit = { has_variable = mnt_rural_goods_value }
		add = { value = var:mnt_rural_goods_value }
	}
}

mnt_location_rural_goods_value_relative = {
	value = mnt_location_rural_goods_value
	divide = owner.var:mnt_top_rural_value_1 # scale relative to top rural producer in owner
}

mnt_location_urban_goods_value_relative = {
	value = mnt_location_urban_goods_value
	divide = owner.var:mnt_top_urban_value_1 # scale relative to top rural producer in owner
}

mnt_location_total_goods_value_relative = {
	value = mnt_location_total_goods_value
	divide = owner.var:mnt_top_total_value_1 # scale relative to top total producer in owner
}
