#Value only used for initial setup. 0.1 * peasants pops in thousands, with minimum of 1 spawned RGO building per Location, and max level of 1/3 of the max level defined for the Location's active raw material
rgo_building_level_precise = {
	value = num_pop_type:peasants
	multiply = 0.1
	max = {
		add = mnt_location_max_rgo_building_level	#Value defined in MnT_rgo_ui_values.txt
		multiply = 0.333
	}
	min = 1
}

#Grabs the value above and round it UP
rgo_building_level = {
	value = rgo_building_level_precise
	ceiling = yes
}

#Grabs the initial setup value, multiplies it by 2 and rounds it UP
rgo_building_level_double = {
	value = rgo_building_level_precise
	multiply = 2
	max = {
		value = mnt_location_max_rgo_building_level
		floor = yes
	}
}

#Max level calculation for the RGO replacement buildings

rgo_alum_max_level = {
	save_temporary_value_as = { 
		name = base_value
		value = 0
	}
	if = { limit = { can_build_alum_RGO = yes }
	    # base
		save_temporary_value_as = { name = base_value
			value = { value = scope:base_value
				add = 0.25
			}
		}
		add = { desc = "BUILDING_LEVEL_BASE"
			value = 0.25
		}
		# dev bonus
		save_temporary_value_as = { name = base_value
			value = { value = scope:base_value
				add = { value = development
					multiply = 0.05
				}
			}
		}
		add = { desc = "BUILDING_LEVEL_DEVELOPMENT"
			value = development
			multiply = 0.05
		}
		# pop bonus
		save_temporary_value_as = { name = base_value
			value = { value = scope:base_value
				add = { value = population
					multiply = 0.01
				}
			}
		}
		add = { desc = "BUILDING_LEVEL_POPULATION"
			value = population
			multiply = 0.01
		}
		min = 1
	}
	if = { limit = { location_rank ?= location_rank:rural_settlement }
		add = { desc = "BUILDING_LEVEL_IS_RURAL_SETTLEMENT"
			value = { value = scope:base_value
				multiply = 0.2
			}
		}
	}
	if = { limit = { raw_material = goods:alum }
		add = { desc = "IDEAL_RGO_AREA"
			value = { value = scope:base_value
				multiply = 2
			}
		}
	}
	if = { limit = { owner ?= { modifier:global_max_rgo_size_modifier != 0 } }
        add = { desc = "GLOBAL_MAX_RGO"
            value = { value = scope:base_value
                multiply = owner.modifier:global_max_rgo_size_modifier
            }
        }
	}
	if = {
	    limit = { 
			OR = {
				location_rank ?= location_rank:town
				location_rank ?= location_rank:city
			}
	        owner ?= { modifier:global_max_rgo_size_modifier_in_non_rural != 0 }
	    }
		add = { desc = "URBAN_MAX_RGO"
			value = scope:base_value
			multiply = owner.modifier:global_max_rgo_size_modifier_in_non_rural
		}
	}
	else_if = { 
		limit = {
			location_rank ?= location_rank:rural_settlement
			owner ?= { modifier:global_max_rgo_size_modifier_in_rural != 0 }
		}
	    add = { desc = "RURAL_MAX_RGO"
			value = scope:base_value
			multiply = owner.modifier:global_max_rgo_size_modifier_in_rural
		}
	}
	if = {
		limit = { modifier:local_raw_material_output != 0 }
		add = {
			desc = "BUILDING_LEVEL_RAW_MATERIALS_BONUS_LOCAL"
			value = { 
				value = scope:base_value
				multiply = modifier:local_raw_material_output
			}
		}
	}
	if = {
		limit = { owner ?= { modifier:global_raw_material_output != 0 } }
		add = {
			desc = "BUILDING_LEVEL_RAW_MATERIALS_BONUS_GLOBAL"
			value = { 
				value = scope:base_value
				multiply = owner.modifier:global_raw_material_output
			}
		}
	}
}
