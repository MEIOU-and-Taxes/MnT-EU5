#The following is used to calculate starting building levels.
rgo_max_bonus = {
	add = {
		desc = "BUILDING_LEVEL_BASE"
		value = 1.2 #rural_bonus
	}
	add = { desc = "BUILDING_LEVEL_DEVELOPMENT"
		value = development
		multiply = 0.1
	}
	add = { desc = "BUILDING_LEVEL_POPULATION"
		value = population
		multiply = 0.025
	}
	if = { limit = { location_rank ?= location_rank:rural_settlement }
		multiply = { desc = "BUILDING_LEVEL_IS_RURAL_SETTLEMENT"
			value = 1.2 #rural_bonus
		}
	}
}

rgo_building_level_precise = {
	value = num_pop_type:peasants
	multiply = 0.1
	max = {
		add = rgo_max_bonus
		multiply = 0.333
	}
}

rgo_building_level = {
	value = rgo_building_level_precise
	ceiling = yes
}

rgo_building_level_double = {
	value = rgo_building_level_precise
	multiply = 2
	ceiling = yes
}
