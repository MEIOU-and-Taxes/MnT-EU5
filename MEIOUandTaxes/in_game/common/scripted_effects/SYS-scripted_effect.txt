print_building_type = {
    set_local_variable = {
        name = bldg_count

        value = {
            every_scope_state = {
                every_scope_building = {
                    limit = { is_building_type = building_$BLDG_TYPE$ }
                    add = level
                }
            }
        }
    }
    if = {
        limit = { local_var:bldg_count > 0 }
        debug_log = "::[GetGlobalVariable('year').GetValue]:[THIS.GetStrategicRegion.GetNameNoFormatting]: $BLDG_TYPE$ = [SCOPE.GetLocalVariable('bldg_count').GetValue]"
    }
}

get_sol = {
    set_local_variable = {
        name = $VAR$

        value = {
            every_scope_state = {
                add = average_sol
                # divide = state_population #TODO: Wrong
            }
        }
    }
    remove_local_variable = $VAR$
}
get_wealth = {
    set_local_variable = {
        name = $VAR$

        value = {
            every_scope_state = {
                every_scope_pop = {
                    add = {
                        value = wealth
                        multiply = total_size
                    }
                }
                divide = state_population
            }
        }
    }
    remove_local_variable = $VAR$
}

count_states_in_strategic_regions = {
    every_in_global_list = {
        variable = strategic_regs
        set_variable = { name = state_count value = 0 }
        every_state = {
            if = {  limit = {  this.region = prev  }  }
            change_variable = { name = state_count add = 1 }
        }
    }
    # Printing
    # set_local_variable = { name = index value = 0 }
    ordered_in_global_list = {
        variable = strategic_regs

        debug_log = "[THIS.GetStrategicRegion.GetNameNoFormatting]: [THIS.Var('state_count').GetValue]"
    }
}
